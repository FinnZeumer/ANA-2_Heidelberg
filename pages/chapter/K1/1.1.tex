\section{Der Funktionen-Raum $\mathcal{R}[a,b]$}
\label{sec:1.1}

\begin{definition}{Riemann-Integrierbarkeit}
    Eine Funktion $f : [a,b] \to \mathbb{C}$ heißt \emph{Riemann-integrierbar} auf $[a,b] \subset \R$,
    falls $\Re(f)$ und $\Im(f)$ Riemann-integrierbar sind.

    Man definiert das Integral komplexwertiger Funktionen durch
    $$
    \int_a^b f(x)dx
    := \int_a^b \Re f(x)dx + \i \int_a^b \Im f(x)dx.
    $$
\end{definition}

Diese Definiton sagt also lediglich aus, dass wenn jeweils der $\Re$-Teil und der $\Im$-Teil einer komplexwertigen Funktion getrennt Rieman-integrierbar sind.
Es ist bekannt, dass die Summe zweier reellwertiger Riemanintegrale immer noch Rieman-integrierbar ist.\footnote{Dies ist keine perfekte Erklärung, da eine komplexe Funktion etwas mehr ist, als lediglich zwei reellwertige Funktionen, jedoch sind sowohl $\Re$, als auch $\Im$ Abbildungen in den $\R$.}

\begin{bemerkungbox}
    \begin{enumerate}
        \item Uneigentliche Riemann-Integrale lassen sich analog für komplexwertige Funktionen definieren.
            Uneigentlich heißt hier, dass das Integral entweder über unbeschrängte Bereiche (e.g. $\pm \infty$) integriert wird, oder eine Singularität auf dem Rand des Definitionsbereich liegt. 
        \item Die Rechenregeln des reellen Riemann-Integrals übertragen sich auf komplexwertige Integrale.
        Insbesondere gilt:
        $$
        \begin{aligned}
        \int_{a}^{b} \overline{f(x)}dx &= \int_{a}^{b} \left(\Re f(x) - \i\Im f(x)\right) dx \\
            &= \underbrace{\int_{a}^{b} \Re f(x) dx}_{\Re\left(\int_{a}^{b}f(x)dx\right)}  -  \i  \underbrace{\int_{a}^{b} \Im f(x) dx}_{\Im\left(\int_{a}^{b}f(x)dx\right)}  \\
            &= \overline{\int_{a}^{b}f(x)dx}
        \end{aligned}
        $$
    \end{enumerate}

    Dabei beschreibt die\afz{overline} die komplexe Konjugation $z = a + \i b, \overline{z} = a - \i b$, und ist in der komplexen Ebene als Spiegelung an der $\Re$-Achse zu verstehen.
\end{bemerkungbox}

\begin{beispielbox}
    Sei $n \in \mathbb{Z}, n \neq 0, f(t) := e^{\i n t}$ und $F'(t) := \frac{1}{\i n} e^{\i n t}$. Dann ist $F'(t) = f(t)$ und daher 
    $$
    \int_{a}^{b} e^{\i n t} dt = \frac{1}{\i n} e^{\i n t} \Big|_{a}^{b} = \frac{1}{\i n} (e^{\i n b} - e^{\i n a}) \, .
    $$   
\end{beispielbox}


\begin{definition}[deflabel=StueckweiseStetigkeit]{Stückweise-Stetigkeit}
    Eine Funktion $f : [a,b] \to \K$ (mit $\K = \mathbb{R}$ oder $\mathbb{C}$) heißt
    \emph{stückweise stetig}, falls:

    \begin{enumerate}
        \item $f$ auf $[a,b]$ beschränkt ist und nur endlich viele Unstetigkeitsstellen besitzt,
        \item an jeder Unstetigkeitsstelle $\xi \in [a,b]$ die links- bzw. rechtswertigen Grenzwerte
        $$
            f(\xi^\pm) := \lim_{h\downarrow 0} f(\xi \pm h)
        $$
        existieren.
    \end{enumerate}

    Für $\xi \in (a,b)$ setzt man zusätzlich die Konvention
    $$
        f(\xi) := \frac{f(\xi^-) + f(\xi^+)}{2}.
    $$
\end{definition}
Diese Definition sagt also aus, dass an einer Stelle $\xi$ die Funktion\afz{unterbrochen} sein kann. Es gibt so mit eigentlich zwei verschiedene Werte an dieser Stelle, einmal denn unteren $\xi^-$, und den oberen $\xi^+$. Diese werden über den Limes bestimmt. Da zwei Funktionswerte für dasselbe Argument hier nicht gewollt sind, wird lediglich das arimetische Mittel der beiden Werte gebildet. 

Solange die Unstetigkeitsstellen jedoch bestimmt werden können -- also abzählbar sind -- ist auch eine Rieman-Integration möglich (solange die Funktion auf dem Integrationsbereich ansonsten stetig ist).


\begin{beispielbox}
    \img[.6]{K1/ANA2__StuckweiseStetig__K1}{Visualisierung von stückweiser-Stetigkeit.}
\end{beispielbox}

\begin{bemerkungbox}
    \begin{itemize}
        \item Stückweise stetige Funktionen sind Riemann-integrierbar.
        \item Die Menge aller stückweise stetigen Funktionen bildet den Vektorraum $\mathcal{R}[a,b]$.
    \end{itemize}
\end{bemerkungbox}

\begin{satz}{Lebesgue (Ana 1)}
    Sei $f:[a,b]\to\mathbb{R}$. Dann ist $f$ Riemann-integrierbar $\Longleftrightarrow$
    $f$ auf $[a,b]$ beschränkt und fast überall stetig.

    Hierbei bedeutet\afz{fast überall}: bis auf eine Nullmenge von Unstetigkeitsstellen.
    \vspace{1em}

    Nullmenge: Eine Menge $N\subset\mathbb{R}$ heißt \emph{Nullmenge}, wenn sie durch endlich viele oder abzählbar unendlich viele Intervalle $(I_k)_{k\in\mathbb{N}}$ überdeckt werden kann derart, dass für jedes $\varepsilon>0$ eine Überdeckung mit
        $$
            N\subset\bigcup_{k=1}^{\infty} I_k
            \qquad\text{und}\qquad
            \sum_{k=1}^{\infty} |I_k| < \varepsilon
        $$
    existiert. Anschaulich bedeutet das, dass die Gesamtlänge der überdeckenden Intervalle beliebig klein gewählt werden kann.
\end{satz}

\begin{definition}[deflabel=Sesquilinearform]{Sesquilinearform}
    Für $f,g \in \cR[a,b]$ definiert man
    $$
        \skalP{f}{g} := \int_a^b f(x)\overline{g(x)}dx.
    $$
    Diese Abbildung ist wohldefiniert, da $f,g \in \cR[a,b]$ und somit auch
    $f\cdot g \in \cR[a,b]$.

    \bigskip
    Äquialente Definition:
    Es eien $V,W$ komplexe Vekotrräume. Eine Abbildung
    $$
        S: V \times W \rightarrow C, \quad (v,w) \mapsto S(v,w) = \langle v, w \rangle
    $$
    heißt \emph{Sesquilinearform}, wenn $S$ linear im ersten und semilinear im zweiten Argument ist.\footnote{Dies kann auch anders herum definiert werden, beide Reihenfolgen sind äquivalent.}
\end{definition}
Diese Definition ist als Skalarprodukt zweier Funktionen zu verstehen. Jedoch gibt es hier wichtige Unterschiede. Denn\afz{sesqui} bedeutet zu deutsch\afz{anderthab}, diese Abbildung ist also nicht vollständig linear. 
Das bedeutet, das die Abbildung im einen Argument linear sein muss, im anderen (zweiten) Argument semilinear. 

Um das Verständnis noch etwas zu vertiefen, ist die Sesquilinearform als Abbildung zwischen zwei Vekotrräumen zu verstehen. Jedes Argument kann aus einem anderen Vektorraum entstammen, müssen jedoch demsselben Skalarkörper $\K$ entstammen. 
Für reellwertige Vekotrräume entspricht die Definiton gerade der Bilinearform. Für komplexwertige Vekotrräume sind die Eigenschaften die oben genannten. 
\begin{beispielbox}
    Das \emph{Standardskalarprodukt} komplexwertiger Funktionen ist eine Sesquilinearform:
    $$
        f : \C^n \times \C^n \to \C
    $$
\end{beispielbox}

\begin{bemerkungbox}
    Eine Kurze Erläuterung zu wichtigen Begriffen:

    Für alle $f,f_1,f_2,g,g_1,g_2 \in \cR[a,b]$ und Skalare $\alpha,\beta \in \C$ gilt:
    \begin{enumerate}
        \item Linearität im ersten Argument:
        $$
            \skalP{\alpha f_1 + \beta f_2}{g}
            = \alpha\skalP{f_1}{g} + \beta \skalP{f_2}{g}
        $$
        \item Semilinearität im zweiten Argument:
        $$
            \skalP{f}{\alpha g_1 + \beta g_2}
            = \overline{\alpha}\skalP{f}{g_1} + \overline{\beta}\skalP{f}{g_2}
        $$
        \item Symmetrieeigenschaft ($\R$ symmetrisch,  $\C$ hermitisch):
        $$
            \skalP{f}{g} = \overline{\skalP{g}{f}}.
        $$
        Für $\K = \R$ gilt somit $\skalP{f}{g} = \skalP{g}{f}$.
        \item Semidefinitheit:
        $$
            \skalP{f}{f} = \int_a^b |f(x)|^2dx \ge 0.
        $$
        \item Aus 4. und \cdef{Sesquilinearform} folgt:
        $$
            \skalP{f}{f} = 0 \Ra f=0 \text{ auf } [a,b]
        $$
        \img{K1/Sesqi_Bem}{Visualisierung zu 5. Nach \cdef{StueckweiseStetigkeit} ist $f(\xi) = 0$, an der Unstetigkeitsstelle, so wie in der roten Abbildung gezeigt. Die blaue Abbildung zeigt jedoch ein Beispiel, indem 5. nicht gilt, dies ist somit keine Sesquilinearform.}
    \end{enumerate}
\end{bemerkungbox}

\begin{definition}{Skalarprodukt}
    Sei $V$ ein Vektorraum über $\K$ (mit $\K = \mathbb{R}$ oder $\mathbb{C}$).
    Eine Abbildung $\skalP{\cdot}{\cdot}: V \times V \to \K$ heißt \emph{Skalarprodukt}, wenn:
    \begin{enumerate}
        \item Symmetrie:
        $$
            \skalP{v}{u} = \overline{\skalP{u}{v}}.
        $$
        \item Linearität:
        $$
            \skalP{\alpha v}{u} = \alpha \skalP{v}{u}, \qquad \skalP{v}{u+w} = \skalP{v}{u} + \skalP{v}{w} \, , \\
            \\
            \skalP{\alpha v}{u} = \overline{\alpha} \skalP{v}{u}, \qquad \skalP{v+u}{w} = \skalP{v}{w} + \skalP{u}{w}.
        $$
        \item Positivdefinitheit:
        $$
            \skalP{v}{v} \ge 0, \quad \skalP{v}{v}=0 \iff v=0.
        $$
    \end{enumerate}

    Das Standardskalarprodukt (euklidische oder kanonische Skalarprodukt) von $x,y \in \R^n$ wird via
    $$
        \langle x,y\rangle := x_1 y_1 + x_2 y_2 + \cdots + x_n y_n = \sum_{i=1}^n x_i y_i = x^T y
    $$
    berechnet. Analog wird es für komplexwertige Funktion $x,y \in \C^n$ via 
    $$
        \langle x,y\rangle := \bar{x}_{1}y_{1} + \bar{x}_{2}y_{2} + \cdots + \bar{x}_{n}y_{n} = \sum_{i=1}^{n} \bar{x}_{i} y_{i} = x^{H} y \, ,
    $$
    oder durch
    $$
        \langle x, y\rangle := x_1 \overline{y}_1 + x_2 \overline{y}_2 + \cdots + x_n \overline{y}_n = \sum_{i=1}^n x_i \overline{y_i} = y^H x
    $$
    berechnet.
\end{definition}
Diese Definition ist als spezialform der Sesquilinearform zu verstehen, denn das Skalarprodukt ist eine Abbildung eines Vektorraumes mit sich selbst auf den Körper, welcher diesen Vektorraum aufspannt. Die Sesquilinearform ist auch für zwei unterschiedlichen Vektorräumen definiert -- das heißt auch, dass jedes Skalarprodukt auch immer eine Sesquilinearform ist. 
Das reelle Skalarprodukt ist dabei trivial und bekannt zu berechnen. 
Das komplexe Standardskalarprodukt ist etwas komplizierter, denn das eine der beiden Arugmente muss komplex konjugiert werden. Als Hinweis; eine komplexe Matrix, die Transponiert und komplex konjugiert wird, heißt hermitisch transponiert und wird mit dem Hochgestellten $H$ gekennzeichnet.

Dazu kommt, dass das Skalarprodukt symmetrisch (hermitisch) sein muss und Positivdefinit. Die (Semi-)Linearität muss für beide gelten. 
Zudem kann jedes Skalarprodukt eine Norm induzieren, für eine Sesquilinearform muss dies nicht gelten. 

\begin{bemerkungbox}
    Positivdefinite symmetrische bilinearform $\K = \R$ \\
    Positivdefinite hermitische sesquilinerform $\K = \C$ \\
    Auf dem Raum $\cR[a,b]$ wird $\skalP{\cdot}{\cdot}$ das $L^2$-Skalarprodukt genannt.
\end{bemerkungbox}

\begin{lemma}{Cauchy--Schwarz-Ungleichung}
    Für alle $f,g \in \cR[a,b]$ gilt die Cauchy--Schwarz-Ungleichung:
    $$
        |\skalP{f}{g}|^2 \le \skalP{f}{f}\skalP{g}{g}.
    $$

    Mit \cdef{L2Norm} kann dies auch als 
    $$
        \left|\langle f, g\rangle\right|^{2} \leq \|f\|^{2} \cdot \|g\|^{2}
    $$
    beschrieben werden.
\end{lemma}
Die Aussage dieses Stazes lässt sich wie folgt verstehen:
$x,y$ sind Elemente eines (also desselben) Vektorraumes ($\K = \R \text{ oder } \C$). Da sheißt die Fläche der Projektion ins Quadrat von $f$ auf $g$ ist kleiner als die multiplizierte Fläche von $f^2$ und $g^2$.
Gleichheit gilt genau dann, wenn $x$ und $y$ linear abhängig sind.
Dies ist sich grob wie die Dreiecksungleichung vorzustellen.

    Im zweidimensonalen reellen Ruam $x,y \in \R^2$ lässt sich dies wie folgt visualisieren:
    \img{K1/ANA2__Cauchy_schwarz_ungleichung__K1}{Visualisierung der Idee der Cauchy-Schwaz-Ungleichung}
\end{beispielbox}

\begin{beweisbox}
    $$
    \begin{aligned}
        &1) \\
         &\text{Für } g \equiv 0 \quad \text{gilt trivialerweise} \\
            &\lvert \skalP{f}{g} \rvert^{2} = 0 = \skalP{f}{f} \cdot \underbrace{\skalP{g}{g}}_{=0}\\[0.5em]
        &2) \\
         &g \not\equiv 0, \quad \alpha \in \mathbb{K}  \text{ bel.}\\
            &0 \le (f + \alpha g , f + \alpha g) = \skalP{f}{f} + \alpha \skalP{g}{f} + \overline{\alpha} \skalP{f}{g} + \underbrace{\alpha \cdot \overline{\alpha}}_{\lvert \alpha \rvert^{2}} \skalP{g}{g}\\[0.5em]
            &\text{Setze} \quad \alpha := - \frac{\skalP{f}{g}}{\skalP{g}{g}} = - \frac{\overline{\skalP{g}{f}}}{\skalP{g}{g}} \\[0.5em]
            &\overline{\alpha} = - \frac{\skalP{g}{f}}{\skalP{g}{g}} \\[0.5em]
            &0 \le \skalP{f}{f} - \frac{\skalP{f}{g}\skalP{g}{f}}{\skalP{g}{g}} - \frac{\skalP{g}{f}\skalP{f}{g}}{\skalP{g}{g}} + \frac{\skalP{f}{g}}{\skalP{g}{g}} \cdot \frac{\skalP{g}{f}}{\skalP{g}{g}}\skalP{g}{g} \\
            &= \skalP{f}{f} - \frac{\skalP{g}{f} \cdot \skalP{f}{g}}{\skalP{g}{g}} + \frac{\skalP{f}{g} \cdot \skalP{g}{f}}{\skalP{g}{g}^{2}}\\
            &= \skalP{f}{f} - \frac{\skalP{g}{f} \cdot \skalP{f}{g}}{\skalP{g}{g}} = \skalP{f}{f} - \frac{\lvert \skalP{f}{g} \rvert^{2}}{\skalP{g}{g}} \\
            &[\skalP{g}{f}\skalP{f}{g} = \overline{\skalP{f}{g}} \cdot \skalP{f}{g} = \lvert \skalP{f}{g} \rvert^{2}]\\[0.5em]
            &\Rightarrow \quad 0 \le \skalP{f}{f} \cdot \skalP{g}{g} - \lvert \skalP{f}{g} \rvert^{2} \quad \text{q.e.d.}\\
    \end{aligned}
    $$
\end{beweisbox}


\begin{definition}[deflabel=L2Norm]{L\textsuperscript{2}-Norm}
    Die durch das Skalarprodukt $\skalP{\cdot}{\cdot}$ induzierte L\textsuperscript{2}-Norm auf $\cR[a,b]$ ist definiert durch
    $$
        \|f\|_{L^2} := \sqrt{\skalP{f}{f}} = \left( \int_a^b |f(x)|^2dx \right)^{1/2} \, ,
    $$
    und erfüllt 
    \begin{enumerate}
        \item positive Semi-Definitheit
        \item Homogenität
        \item die Dreiecksungleichung
    \end{enumerate}

    \bigskip

    Notation:
    Sei $V$ ein Vektorraum und $a\in V$, so bezeichnet $\left\|a\right\|$ eine Norm von $a$
\end{definition}
Die $L^2$ norm ist eine besondere $L^p$-Norm (nach \cdef{LpNorm}), denn dies ist die einzige jener Normen, die den euklidischen Raum bezeichnet und ist die einzige $L^p$-Norm, welche sich mit einem Skalarprodukt versehenlässt. Der $L^2$ wird auch\afz{Hilbertraum} genannt.
Als Randnotiz: Der $L^2$ ist selbst-dual.
Anschaulich beschreibt dies die Länge eines Vektors, bzw. den Betrag dessen.

\begin{bemerkungbox}
    Die L\textsuperscript{2}-Norm erfüllt:
    \begin{enumerate}
        \item Definitheit: $\|f\|_2 = 0 \Ra f = 0$.
        $$
            \lVert f \rVert = 0 \Ra \skalP{f}{f} = 0 \implies f \equiv 0 \quad \text{auf } [a,b], \text{ da } f\in \cR[a,b] \quad
            (\text{weil } f \in R[a,b])
        $$
        Da das Skalarprodukt positivdefinit ist, "erben" normen diese Eigenschaft, da die Wruzel eines positivdefiniten Wertes positivdefinit bleibt. Ein Skalarprodukt mit sich selbst kann nur null sein, wenn das Argument selbst das Nullelement des Vektorraumes ist.

        \item Homogenität: $\|\alpha f\|_2 = |\alpha|\|f\|_2$.
        $$
            \lVert \alpha f \rVert = \sqrt{\skalP{\alpha f}{\alpha f}}\\
            = \sqrt{\lvert \alpha \rvert^2 \skalP{f}{f}} = \lvert \alpha \rvert \sqrt{\skalP{f}{f}} = \lvert \alpha \rvert \cdot \lVert f \rVert\\[0.5em]
        $$
        Heißt, der Vekotr kann erst um $\alpha$ skaliert werden und dann die Norm des skalierten Vekotrs bestimmt werden oder es wird zuerst die Norm auf $f$ angewendet und dann um $\alpha$ skaliert. Beides bestimmt den Betrag (die Länge) des skalierten Vektors.

        \item Dreiecksungleichung $\|f+g\|_2 \le \|f\|_2 + \|g\|_2$.:
        $$
            \begin{aligned}
                \lVert f + g \rVert &= \skalP{f+g}{f+g}^{\frac{1}{2}} \\
                & = (\lVert f \rVert^2 + \skalP{f}{g} + \skalP{g}{f} + \lVert g \rVert^2)^{\frac{1}{2}} \\[0.5em]
                & \quad \text{Nach Lemma 6: C.S.-Ungl.} \\
                &\leq   (\lVert f \rVert^2 + 2 \lVert f \rVert \lVert g \rVert + \lVert g \rVert^2)^{\frac{1}{2}}\\
                & = \sqrt{(\lVert f \rVert + \lVert g \rVert)^2} = \lVert f \rVert + \lVert g \rVert
            \end{aligned}
        $$
        \end{enumerate}
\end{bemerkungbox}


\begin{definition}{L\textsuperscript{2}-Konvergenz}
    Seien $f_n \in \cR[a,b], n \in \N \text{ und } f\in \cR[a,b]$ gegeben.
    Eine Folge $(f_n)_{n\in\N}$ konvergiert gegen $f$ im \emph{quadratischen Mittel}
    $f_n \underset{{L^2}}{\to} f$, wenn
    $$
        \|f_n - f\|_{L^2} \longrightarrow 0 \quad n \to \infty.
    $$

    d.h. das quadratische Mittel der Abweichung zwischen $f_n$ und $f$ geht gegen Null:
    $$
        \|f_n - f\|_{L^2}^2 = \int_a^b |f_n(x) - f(x)|^2dx \to 0  \Lra \|f_n - f\|_{L^2} \to 0.
    $$
\end{definition}
Grundsätzlich heißt Konvergenz lediglich, dass sich eine Reihe immer näher an einen Wert annährt. 
\img[.5]{K1/ANA2__l2Stetigkeit__K1}{Grundidee der (Punktweisen-)Konvergenz.}
Im Fall der $L^2$ Konvergenz, sollte man sich $f$ als eine Funktion vorstellen, gegenwelche die Reihe von Funktionen $f_n$ konvegiert. In der Definiton der $L^2$-Norm befindet sich ein integral eines positivdefiniten Wertes. Dies ist hier gerade als Fläche zwischen zwei Funktionen zu verstehen. 
Konvergiert nun also eine Funktionenreihe $f_n$ gegen $f$, so heoßt dies lediglich, dass die Fläche zwischen diesen Funktion gegen null geht. 
\img[.8]{K1/L2_Konvergenz_Flache}{Visualisierung der $L^2$-Konvergenz über die Fläche zwischen zwei Funktionen.}

\begin{bemerkungbox}
    Für jede Folge $f_n$ gilt:
    $$
        \|f_n - f\|_2^2 \le (b-a)\|f_n - f\|_\infty^2 \, ,
    $$

    denn es gilt:
    $$
        \left|f_n(x) - f(x)\right|^2 \leq \left(\max_{x \in [a,b]} \left|f_n(x) - f(x)\right|\right)^2 = \left\|f_n - f\right\|_\infty^2
    $$

    Daraus folgt:
    $$
        \|f_n - f\|_\infty^2 \to 0 \Rightarrow \|f_n - f\|_2^2 \to 0, \qquad n \to \infty
    $$
    aber die Umkehrung gilt im Allgemeinen nicht.   
\end{bemerkungbox}

\begin{beispielbox}
    Betrachte $f_n(x) := x^n$ auf $[-1,1]$. Dann gilt:
    $$
    \begin{aligned}
        \|f_n\|_{L^2}^2 
        &= \int_{-1}^{1} x^{2n} \, dx 
        = 2 \int_0^1 x^{2n} \, dx \\ 
        &= 2 \frac{x^{2n+1}}{2n+1} \Bigg|_0^1 
        = \frac{2}{2n+1} \longrightarrow 0, \quad n \to \infty \\ 
        &\Rightarrow f_n \underset{L^2}\to f \equiv 0, \quad n \to \infty
    \end{aligned}
    $$
    also $f_n \to 0$ in L$^2$.
    
    \img[0.7]{K1/Beispiel_L2_Konvergenz}{Visualisierung des Beispiels für $n \in \{1,2,3\}$.}

    Für $x=1$ gilt:
    $$
        f_n(1) = 1 \quad \forall n \in \N,
    $$

    $
        \Rightarrow \text{keine punktweise Konvergenz} \quad f_n(x) \not\to 0 \quad \forall x\in [-1,1], \quad n \to \infty
    $

    $
        \Rightarrow \text{keine gleichmäßige Konvergenz} \quad \left\| f_n - f \right\|_\infty \not \to  0, \quad f\equiv 0
    $

    Für $x=-1$ gilt:
    $$
        f_n(-1) = (-1)^n \quad \forall n \in \N,
    $$

    $
        \Rightarrow f_n(-1) \text{ divergiert.}
    $
\end{beispielbox}

\begin{definition}[deflabel=LpNorm]{L\textsuperscript{p}-Norm}
    Für $p \in [1,\infty)$ und $f : [a,b] \to \mathbb{R}$ definiert man:
    $$
        \|f\|_p := \left( \int_a^b |f(x)|^pdx \right)^{1/p}.
    $$
    Für $p=2$ erhält man die L$^2$-Norm.
\end{definition}
Dies ist die allgemeine Definition der $L^p$ oder häufig auch $\ell^p$-Normen, wobei die $L^2$-Norm (\cdef{L2Norm}) lediglich eine Spezialform darstellt.
Jeder $L^p$-Raum ist auch ein normierter Raum (\cdef{NormRaum}).

\begin{bemerkungbox}
    \begin{enumerate}
        \item Der Raum $\cR[a, b]$ mit $L^2$-Norm $\left\|\cdot\right\|$ ist nicht vollständig, d. h., es existieren Cauchy-Folgen in $\cR[a, b]$ die keinen Limes in $\cR[a, b]$ haben.
        \item Man benötigt einen “Vervollständigungsprozess” über die Äquivalenzklassen von Cauchy-Folgen.
        $\Rightarrow$ Der $L^2 [a,b]$-Raum wird eingeführt.
        \item In diesem Sinne ist die $L^2$ -Norm eine Semi-Norm (sie ist nicht streng definit).
    \end{enumerate}
    \vspace{1em}

    \textbf{Erinnerung}:

    $C[a,b]$ Raum stetiger Funktionen mit
    $$
    \|f\|_{\infty} := \max_{x\in[a,b]} |f(x)| \quad\text{(Maximumsnorm)}
    $$

    $C[a,b]$ ist vollständig $\iff$ C.F. $(f_n)$ haben einen Limes $f$ in $C[a,b]$.

    \bigskip

    $\|\cdot\|_{L^2}$ ist nicht streng definiert:

    $$
    \|f\|_{L^2}=0 \quad\Longleftrightarrow\quad f\equiv 0 \text{ auf } [a,b]
    $$

    $\Longrightarrow$ $f\equiv 0$ gilt nur, wenn $f\in \mathcal{R}[a,b]$,

    aber nicht für z.\,B. allgemeine, unstetige Funktionen.

    \img[.8]{K1/ANA2__StrengDefinit__K1}{Allgemeine unstetige Funktion.}


    Falls $f\in \mathcal{R}[a,b]$, dann gilt (für einen Punkt $\xi$ mit beidseitigen Grenzwerten)
    $$
    f(\xi)=\frac{f(\xi_-)+f(\xi_+)}{2}=\frac{0+0}{2}=0.
    $$
\end{bemerkungbox}

\subsection*{Geometrie im $\cR [a,b]$}
\begin{definition}{Orthogonalität}
    Für $f,g \in R[a,b]$ heißt $f$ orthogonal zu $g$, wenn
    $$
        \skalP{f}{g} = 0.
    $$
    Eine Menge $S \subset R[a,b]$ heißt \emph{orthogonal}, wenn für alle
    $f_i,f_j \in S$:
    $$
        \skalP{f_i}{f_j} =
        \begin{cases}
            0, & i \ne j,\\
            \|f_i\|_2^2, & i = j.
        \end{cases}
    $$
\end{definition}

\begin{beispielbox}
    Betrachte die Funktionen $f(x) = x$ und $g(x) = x^2$ auf dem Intervall $[a,b] = [-1, 1]$.   

    Über die Definition der $L^2$-Norm ergibt sich deren Skalarprodukt zu
    $$
    \begin{aligned}
        \langle f,g\rangle &= \int_{-1}^{1} x \cdot x^{2} \, dx = \int_{-1}^{1} x^{3} \, dx \\
        &= \frac{1}{4} x^4 \Big|_{-1}^1 = \frac{(1)^4}{4} - \frac{(-1)^4}{4} = \frac{1}{4} - \frac{1}{4} = 0 \, .
    \end{aligned}
    $$
    \img[1]{K1/Orthogonale_Funktionen}{Visualisierung des Beispiels für Orthogonale Funktionen.}
\end{beispielbox}

\begin{satz}{Orthogonalität trigonometrischer Funktionen}
    Die trigometrische Funktionen
    $$
        c_0(x)=1,\quad c_k(x)=\cos(kx),\quad s_l(x)=\sin(lx),\quad k,l\in\mathbb{N},
    $$
    bilden auf $\cR[0,2\pi]$ ein Orthogonalsystem bezüglich des $L^2$-Skalarprodukts.
    Es gilt:
    $$
        \int_{0}^{2\pi} c_k(x) \, dx = \int_{0}^{2\pi} s_l(x) \, dx = \int_{0}^{2\pi} c_k(x) s_l(x) \, dx = 0
    $$
    $$
        \int_{0}^{2\pi} c_k(x)c_l(x) \, dx = \pi \delta_{kl}\, , \qquad
        \int_{0}^{2\pi} s_k(x)s_l(x) \, dx = \pi \delta_{kl}    
    $$

    Dabei beschreibt
    $$
        \delta_{kl} = \begin{cases}1,& l=k\\0,& l\neq k\end{cases}
    $$
    Das Kronecker-Symbol (Kronecker-Delta).
\end{satz}

\begin{beweisbox}
    \begin{align*}
        \int_{0}^{2\pi} C_k(x)\,dx &= \int_{0}^{2\pi} \cos(kx)\,dx
        = \frac{1}{k}\sin(kx)\Big|_{0}^{2\pi}=0 \quad (k\neq 0),\\[10pt]
        \int_{0}^{2\pi} S_k(x)\,dx &= \int_{0}^{2\pi} \sin(kx)\,dx
        = -\frac{1}{k}\cos(kx)\Big|_{0}^{2\pi}
        = -\frac{1}{k}\big(\underbrace{\cos(2\pi k)-\cos 0}_{1-1}\big)=0.
    \end{align*}

    Partielle Integration:
    $$
        \int_a^b u'(x)v(x)\,dx = u(x)v(x)\Big|_a^b - \int_a^b u(x)v'(x)\,dx.
    $$

    Orthogonalität von Cosinus und Sinus:
    \begin{enumerate}
        \item Cosinus mit Sinus:
            \begin{align*}
                \int_{0}^{2\pi} C_k(x)S_\ell(x)\,dx
                &= \int_{0}^{2\pi} \underbrace{\cos(kx)}_{u\prime} \underbrace{\sin(\ell x)}_{v}\,dx\\
                &= \underbrace{\frac{1}{k} \, \sin(kx) \cdot \sin(\ell x) \Big|_{0}^{2\pi}}_{=0} - \int_{0}^{2\pi} \frac{1}{k} \sin(kx) \cos(\ell x) \, dx \\
                &= -\frac{\ell}{K} \int_0^{2\pi} \sin(kx) \, \cos(\ell x) \, dx \\
                &= -\frac{\ell}{K} \int_0^{2\pi} S_k(x) \, C_\ell(x) \, dx \\
                &\left(= \tfrac{1}{2}\int_{0}^{2\pi}\big(\sin((\ell+k)x)-\sin((\ell-k)x)\big)\,dx = 0\right) \, ,
            \end{align*}
            da das Integral von Sinus über ein ganzzahliges Vielfaches von $2\pi$ verschwindet.
            Also gilt für alle $k,\ell\in\mathbb{Z}_{\ge0}$:
            $$
            \int_{0}^{2\pi} C_k(x)S_\ell(x)\,dx = 0.
            $$

        \item Cosinus mit Cosinus:
            \begin{align*}
                \int_{0}^{2\pi} C_k(x)C_\ell(x)\,dx
                &= \int_{0}^{2\pi} \underbrace{\cos(kx)}_{u\prime} \underbrace{\cos(\ell x)}_{v}\,dx\\
                &= \tfrac{1}{2}\int_{0}^{2\pi}\big(\cos((k-\ell)x)+\cos((k+\ell)x)\big)\,dx.
            \end{align*}
            - Falls $k\neq\ell$, beide Kosinus-Frequenzen sind nicht konstant und ihre Integrale über $[0,2\pi]$ verschwinden, also
            $$
            \int_{0}^{2\pi} C_k C_\ell = 0 \quad (k\neq\ell).
            $$
            - Falls $k=\ell\neq0$, dann $\cos((k-\ell)x)=\cos 0=1$ und $\cos((k+\ell)x)=\cos(2kx)$ liefert
            $$
            \int_{0}^{2\pi} C_k^2(x)\,dx = \tfrac{1}{2}\int_{0}^{2\pi}1\,dx + \tfrac{1}{2}\int_{0}^{2\pi}\cos(2kx)\,dx
            = \tfrac{1}{2}\cdot 2\pi + 0 = \pi.
            $$
            - Für $k=\ell=0$ gilt offensichtlich $\int_{0}^{2\pi}1\,dx=2\pi$.

        \item Sinus mit Sinus (analog):
        $$
        \int_{0}^{2\pi} S_k(x)S_\ell(x)\,dx =
        \begin{cases}
            0, & k\neq\ell,\\[4pt]
            \pi, & k=\ell\neq0,\\[4pt]
            0, & k=\ell=0\ (\text{trivial, }S_0\equiv 0).
        \end{cases}
        $$
    \end{enumerate}

    Zusammenfassung:

    - Für $k\neq\ell$: $\displaystyle\int_{0}^{2\pi} C_k C_\ell\,dx = \int_{0}^{2\pi} S_k S_\ell\,dx = \int_{0}^{2\pi} C_k S_\ell\,dx = 0.$

    - Für $k=\ell\neq0$: $\displaystyle\int_{0}^{2\pi} C_k^2\,dx = \int_{0}^{2\pi} S_k^2\,dx = \pi.$

    - Für $k=\ell=0$: $\displaystyle\int_{0}^{2\pi} C_0^2\,dx = 2\pi$ und $S_0\equiv0$.

    Analog für Sinus. q.e.d.

    \img[1]{K1/Orthos_alle}{Visualisierung der verschiedenen Fälle. Rote und grüne Flächen haben dabei immer denselben Betrag, blaue Flächen sind ungleichung null.}
\end{beweisbox}


\begin{definition}{Periodische Funktion}
    Eine Funktion $f:\mathbb{R}\to \K$ heißt $L$-periodisch ($L>0$), wenn
    $$
        f(x+L)=f(x)\quad\forall x\in\mathbb{R}.
    $$
\end{definition}

\begin{beispielbox}
    Sei $f$ $L$-periodisch und $p>0$. Dann sei $\tilde f(x)=f\!\bigl(\tfrac{L}{p}x\bigr)$.

    Für $p=2\pi$ und $\tilde f(x)=f\!\bigl(\tfrac{L}{2\pi}x\bigr)$ ist $\tilde f$ $2\pi$-periodisch:
    $$
    \begin{aligned}
    \tilde f(x+2\pi)
    &= f\!\bigl(\tfrac{L}{2\pi}(x+2\pi)\bigr) \\
    &= f\!\bigl(\tfrac{L}{2\pi}x + L\bigr) \\
    &= f\!\bigl(\tfrac{L}{2\pi}x\bigr) \\
    &= \tilde f(x),
    \end{aligned}
    $$
    da $f$ $L$-periodisch ist.

    \smallskip

    \noindent\emph{Anmerkung:} Variablentransformation; $\tilde f:\mathbb{R}\to\mathbb{K}$, $p$-periodisch.

    \bigskip

    \begin{itemize}
        \item Deshalb betrachten wir nur $2\pi$-periodische Funktionen $f:\mathbb{R}\to\mathbb{K}$.
        \item Wir sagen, dass eine $2\pi$-periodische Funktion $f$ zu $\mathcal{R}[0,2\pi]$ gehört, falls $f$ auf $[0,2\pi]$ Riemann-integrierbar ist.
    \end{itemize}

    
    \textbf{Beispiel einer periodischen Funktion}

    \img{K1/ANA2__Linear_Func_periodisch__K1}{Visualisierung einer beispielhaften periodischen Funktion. $f\in \cR[0,2\pi]$}  
    $$
        f(0)=f(2\pi) \quad  \text{Periodizität!} \\
        \\
        \Rightarrow f(0)=\frac{f(0_-)+f(0_+)}{2}=\frac{2\pi-2\pi}{2}=0
    $$

    \textbf{Trigonometrische Polynome}

    Für $a_k,b_k\in\mathbb{C}$ betrachten wir
    $$
    \boxed{\,f_n(x)=\frac{a_0}{2}+\sum_{k=1}^n\bigl(a_k\cos(kx)+b_k\sin(kx)\bigr)\,}
    $$

    Es gelten die Identitäten
    $$
    \cos x=\tfrac{1}{2}\bigl(e^{ix}+e^{-ix}\bigr),\qquad
    \sin x=\tfrac{1}{2i}\bigl(e^{ix}-e^{-ix}\bigr),\qquad
    \frac{1}{i}=-i.
    $$

    Mit diesen erhält man
    $$
    \begin{aligned}
    f_n(x)
    &=\frac{1}{2}\left(a_0+\sum_{k=1}^n (a_k-i b_k)e^{ikx}+\sum_{k=1}^n (a_k+i b_k)e^{-ikx}\right)\\
    &=\sum_{k=-n}^{n} c_k e^{ikx},
    \end{aligned}
    $$
    wobei die Koeffizienten $c_k$ durch
    $$
    c_k=
    \begin{cases}
    \displaystyle \tfrac{1}{2}\bigl(a_k-i b_k\bigr), & k>0,\\[6pt]
    \displaystyle \tfrac{a_0}{2}, & k=0,\\[6pt]
    \displaystyle \tfrac{1}{2}\bigl(a_{-k}+i b_{-k}\bigr), & k<0,
    \end{cases}
    $$
    gegeben sind.

    Insbesondere (für $k>0$) gilt die Umkehrbeziehung
    $$
    a_k = c_k + c_{-k},\qquad b_k = i\bigl(c_k - c_{-k}\bigr).
    $$

    Beobachtung: Wenn $f$ ein trigonometrisches Polynom ist, sind die Koeffizienten eindeutig und
    $$
    a_k=\frac{1}{\pi}\int_{0}^{2\pi} f(x)\cos(kx)\,dx,\qquad
    b_k=\frac{1}{\pi}\int_{0}^{2\pi} f(x)\sin(kx)\,dx.
    $$

    Für die komplexen Koeffizienten gilt
    $$
    c_k=\frac{1}{2\pi}\int_{0}^{2\pi} f(x)e^{-ikx}\,dx.
    $$
    Man kann dies auch als Skalarprodukt-Formel schreiben (mit dem üblichen Skalarprodukt auf $L^2([0,2\pi])$):
    $$
    c_k=\frac{1}{2\pi}\bigl(f,\;e^{ikx}\bigr)=\frac{1}{2\pi}\int_{0}^{2\pi} f(x)\,\overline{e^{ikx}}\,dx.
    $$

    \begin{beweisbox}
        Zuerst für $\lambda\in\mathbb{R},\ \lambda\neq0$:
        $$
        \int_a^b e^{i\lambda x}\,dx
        =\int_a^b \cos(\lambda x)\,dx + i\int_a^b \sin(\lambda x)\,dx
        =\frac{1}{i\lambda}e^{i\lambda x}\Big|_a^b.
        $$
        Daraus folgt insbesondere für $k\in\mathbb{Z}\setminus\{0\}$:
        $$
        \int_0^{2\pi} e^{ikx}\,dx = 0.
        $$

        Dann
        $$
        \int_0^{2\pi} e^{ik_1 x}\,e^{-ik_2 x}\,dx
        =\int_0^{2\pi} e^{i(k_1-k_2)x}\,dx
        =\begin{cases}
        0, & k_1\neq k_2,\\[4pt]
        2\pi, & k_1=k_2.
        \end{cases}
        $$
        (Orthogonalitätsrelation der Exponentialfunktionen.)

        Angenommen
        $$
        f(x)=\sum_{k=-n}^{n} c_k e^{ikx}.
        $$
        Betrachte das Skalarprodukt $\langle f,e^{i\ell x}\rangle=\int_0^{2\pi} f(x)e^{-i\ell x}\,dx$. Dann
        \begin{align*}
        \int_0^{2\pi} f(x)e^{-i\ell x}\,dx
        &=\int_0^{2\pi}\Big(\sum_{k=-n}^{n} c_k e^{ikx}\Big)e^{-i\ell x}\,dx
        =\sum_{k=-n}^{n} c_k \int_0^{2\pi} e^{i(k-\ell)x}\,dx\\
        &=2\pi c_\ell.
        \end{align*}
        Also
        $$
        c_k=\frac{1}{2\pi}\int_0^{2\pi} f(x)e^{-ikx}\,dx\qquad (k\in\mathbb{Z}).
        $$

        Für die reellen Koeffizienten $a_k,b_k$ gilt (für $k\ge1$):
        $$
        a_k=\frac{1}{\pi}\int_0^{2\pi} f(x)\cos(kx)\,dx,\qquad
        b_k=\frac{1}{\pi}\int_0^{2\pi} f(x)\sin(kx)\,dx.
        $$
        Mit $\cos(kx)=\tfrac{1}{2}(e^{ikx}+e^{-ikx})$ und $\sin(kx)=\tfrac{1}{2i}(e^{ikx}-e^{-ikx})$ erhalten wir
        \begin{align*}
        a_k
        &=\frac{1}{\pi}\int_0^{2\pi} f(x)\frac{e^{ikx}+e^{-ikx}}{2}\,dx
        =\frac{1}{2\pi}\Big(\int_0^{2\pi} f(x)e^{ikx}\,dx+\int_0^{2\pi} f(x)e^{-ikx}\,dx\Big)\\
        &=c_{-k}+c_k,\\[6pt]
        b_k
        &=\frac{1}{\pi}\int_0^{2\pi} f(x)\frac{e^{ikx}-e^{-ikx}}{2i}\,dx
        =\frac{1}{2\pi i}\Big(\int_0^{2\pi} f(x)e^{ikx}\,dx-\int_0^{2\pi} f(x)e^{-ikx}\,dx\Big)\\
        &=\frac{c_{-k}-c_k}{i}=i\big(c_k-c_{-k}\big).
        \end{align*}
        Daraus folgt (für $k\ge1$)
        $$
        c_k=\tfrac{1}{2}\big(a_k - i b_k\big),\qquad
        c_{-k}=\tfrac{1}{2}\big(a_k + i b_k\big),
        $$
        und für die Gleichung des Mittelwerts $c_0=\tfrac{a_0}{2}$.

        Somit gilt die Entsprechung der Reihen:
        $$
        f(x)=\frac{a_0}{2}+\sum_{k=1}^\infty\big(a_k\cos(kx)+b_k\sin(kx)\big)
        =\sum_{k=-\infty}^{\infty} c_k e^{ikx}.
        $$
    \end{beweisbox}
\end{beispielbox}
