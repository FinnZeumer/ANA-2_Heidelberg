\section{Totale Ableitungen}
\label{4.2}

\begin{definition}{Totale Ableitung}
    Es sei $D \subset \R^n$ offen und $f : D \to \R^m$ eine Abbildung.
    $f$ heiĂźt im Punkt $x \in D$ \emph{(total) differenzierbar}, falls es eine lineare Abbildung $A : \R^n \to \R^m$ existiert, sodass
    $$
        \lim_{h\to 0 \atop h\neq 0} \frac{f(x+h) - f(x) - A \cdot h}{\|h\|} = 0 \, .
    $$
    $A$ heiĂźt das \emph{Differential} von $f$ im Punkt $x$.
    Schreibweise:
    $$
        \mathrm{d}f(x), \mathrm{d}f\big|_{x}, \mathrm{d}f_x, Df(x), Df\!\big|, Df_x, \mathrm{d}f(x)\big|_{x=x_0}, Df(x_0) \, .
    $$
\end{definition}

\begin{satz}{Differenzierbarkeit}
    Sei $D\subset\mathbb{R}^n$ offen. FĂĽr Funktionen $f\colon D\to\mathbb{R}^m$ gilt:

    \begin{enumerate}
    \item Sei $f$ in $x\in D$ differenzierbar, dann ist $f$ partiell differenzierbar und
    $$
    Df(x)=J_f(x),
    $$
    wobei $J_f(x)$ die Jacobi-Matrix ist.
    \item Sei $f$ partiell differenzierbar in einer Umgebung von $x\in D$ und die partiellen Ableitungen stetig in $x$, dann ist $f$ differenzierbar in $x$.
    \end{enumerate}   
\end{satz}

\begin{korollar}{}
    stetig partiell differenzierbar $\Longrightarrow$ (total) differenzierbar (und somit stetig) $\Longrightarrow$ partiell differenzierbar.

    Die umgekehrten Implikationen gelten im Allgemeinen nicht.   
\end{korollar}

\begin{satz}{Kettenregel}
    Seien $D_f\subset\mathbb{R}^n$ und $D_g\subset\mathbb{R}^m$ offen, $g: D_g\to\mathbb{R}^n$, $f: D_f\to\mathbb{R}^r$ Abbildungen.
    \vspace{1em}

    Falls $g$ im Punkt $x\in D_g$ und $f$ im Punkt $y=g(x)\in D_f$ differenzierbar sind, gilt:
    \vspace{1em}

    Die Komposition $h=f\circ g$ ist im Punkt $x$ differenzierbar und

    $$
    \underbrace{\overset{\text{Jacobi-Matrix von }f}{D_y f(g(x))}}_{\in\mathbb{R}^{\,r\times n}}
    \;\cdot\;
    \underbrace{\overset{\text{Jacobi-Matrix von }g}{D_x g(x)}}_{\in\mathbb{R}^{\,n\times m}}
    \;=\;
    D_x h(x)\in\mathbb{R}^{\,r\times m}.
    $$

\end{satz}

\begin{definition}{Richtungsableitung}
    Sei $D \in \R^n$ offen, $f : D \to R$ im Punkt $x \in D$ differenzierbar. $f$ ist eine skalare Funktion.
    Dann gilt $\forall v \in \R^n$ mit $\left\|v\right\|^2 = 1$ existiert die Ableitung in Richtung $v$ (sog. \emph{Richtungsableitung})
    $$
        \frac{\partial f}{\partial v}(x) : = \lim_{t \downarrow 0} \frac{f(x + t v) - f(x)}{t} \, .
    $$
\end{definition}

\begin{lemma}{Richtungsableitung}
    Sei $D \in \R^n$ offen, $f : D \to R$ im Punkt $x \in D$ differenzierbar. $f$ ist eine skalare Funktion.
    Dann gilt $\forall v \in \R^n$ mit $\left\|v\right\|^2 = 1$ existiert die Ableitung in Richtung $v$ (sog. \emph{Richtungsableitung})
    $$
        \frac{\partial f}{\partial v}(x) = (\nabla f(x), v)_2 = \nabla f(x)^T v \, .
    $$
\end{lemma}
