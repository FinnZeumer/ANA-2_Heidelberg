\section{Taylor-Entwicklung}
\label{4.4}

\subsection*{Erinnerung -- HĂ¶here Ableitungen}

\begin{enumerate} %[label=(\arabic*)]
    \item Sei $D\subset\mathbb{R}^n$ offen, $f\colon D\to\mathbb{R}^m$ partiell differenzierbar. FĂĽr $j\in\{1,\dots,n\}$ bezeichne
    $$
    \partial_j f=\frac{\partial f}{\partial x_j}\colon D\to\mathbb{R}^m,
    $$
    und die Ableitungsabbildung (Jacobi-Matrix)
    $$
    Df=\big(\partial_1 f,\dots,\partial_n f\big)\colon D\to\mathbb{R}^{m\times n}.
    $$
    Ist nun jede $\partial_j f$ wieder partiell differenzierbar, so sind die zweiten partiellen Ableitungen definiert durch
    $$
    \partial_i\partial_j f=\frac{\partial^2 f}{\partial x_i\partial x_j},\qquad i,j\in\{1,\dots,n\}.
    $$

    Allgemein (induktiv): $f\colon D\to\mathbb{R}^m$ ist $(k+1)$-mal partiell differenzierbar, genau dann, wenn
    \begin{enumerate}%[label=(\roman*)]
        \item $f$ ist $k$-mal partiell differenzierbar, und
        \item alle partiellen Ableitungen der $k$-ten Ordnung sind wieder partiell differenzierbar, d.\,h. fĂĽr jede Folge $i_1,\dots,i_k\in\{1,\dots,n\}$ gilt, dass
        $$
        \partial_{i_1}\partial_{i_2}\dots\partial_{i_k} f
        $$
        partiell differenzierbar ist.
    \end{enumerate}

    \item $f\colon D\to\mathbb{R}^m$ ist $k$-mal stetig partiell differenzierbar, wenn
    \begin{enumerate}%[label=(\roman*)]
        \item $f$ ist $k$-mal partiell differenzierbar, und
        \item alle partiellen Ableitungen der $k$-ten Ordnung sind stetige Abbildungen auf $D$.
    \end{enumerate}

    Bezeichnung: $f\in C^k(D,\mathbb{R}^m)$.

    \item Es gilt:
    $$
    f\in C^1(D,\mathbb{R}^m)\quad\Longleftrightarrow\quad
    \frac{\partial f}{\partial x_i}:D\to\mathbb{R}^m\ \text{ist stetig}\ \forall i\in\{1,\dots,n\}.
    $$
    FĂĽr $m=1$ schreibt man entsprechend
    $$
    \frac{\partial f}{\partial x_i}:D\to\mathbb{R}\ \text{ist stetig}\ \forall i\in\{1,\dots,n\}.
    $$
    Ă„quivalent: $f$ ist total differenzierbar in $D$ und stetig, und die Jacobi-Matrix
    $$
    x\mapsto J_f(x)=\
    \begin{pmatrix}
        \frac{\partial f_1}{\partial x_1}(x) & \cdots & \frac{\partial f_1}{\partial x_n}(x)\\[4pt]
        \vdots & & \vdots\\[4pt]
        \frac{\partial f_m}{\partial x_1}(x) & \cdots & \frac{\partial f_m}{\partial x_n}(x)
    \end{pmatrix}
    $$
    ist stetig als Abbildung $D\to\mathbb{R}^{m\times n}$.

    \item Ist $f\in C^{k+1}(D,\mathbb{R}^m)$, dann ist jede partielle Ableitung der $(k+1)$-ten Ordnung
    $$
    \frac{\partial^{k+1} f}{\partial x_{i_1}\cdots\partial x_{i_{k+1}}}\colon D\to\mathbb{R}^m
    $$
    total differenzierbar (da stetig partiell differenzierbar) und somit stetig. Induktiv folgt: alle partiellen Ableitungen $j$-ter Ordnung mit $j\le k+1$ sind total differenzierbar und stetig auf $D$.

    \item $C^\infty(D,\mathbb{R}^m)=\bigcap_{k=0}^\infty C^k(D,\mathbb{R}^m).$

    Es gilt die Kettenfolge von Inklusionen
    $$
    C^\infty(D,\mathbb{R}^m)\subset\cdots\subset C^{k+1}(D,\mathbb{R}^m)\subset C^k(D,\mathbb{R}^m)\subset\cdots\subset C^1(D,\mathbb{R}^m)\subset C^0(D,\mathbb{R}^m).
    $$

    \item Sei $D\subset\mathbb{R}^n$ offen, $f\colon D\to\mathbb{R}$. Falls die partiellen Ableitungen
    $$
    \frac{\partial^2 f}{\partial x_i\partial x_j},\quad \frac{\partial^2 f}{\partial x_j\partial x_i}\quad (i,j\in\{1,\dots,n\})
    $$
    auf $D$ existieren und $\frac{\partial^2 f}{\partial x_i\partial x_j}$ stetig ist in $x\in D$, dann existiert $\frac{\partial^2 f}{\partial x_j\partial x_i}$ und es gilt
    $$
    \frac{\partial^2 f}{\partial x_i\partial x_j}(x)=\frac{\partial^2 f}{\partial x_j\partial x_i}(x)\qquad\forall x\in D.
    $$

    \item Seien $D\subset\mathbb{R}^n$ offen und $f\in C^k(D,\mathbb{R}^m)$. Sei $\pi$ eine Permutation von $(i_1,\dots,i_k)$. Dann gilt
    $$
    \partial_{i_1}\partial_{i_2}\dots\partial_{i_k} f=
    \partial_{i_{\pi(1)}}\partial_{i_{\pi(2)}}\dots\partial_{i_{\pi(k)}} f.
    $$
    (Vertauschung der Reihenfolge der partiellen Ableitungen ist mĂ¶glich, wenn die benĂ¶tigten Ableitungen existieren und stetig sind.)
\end{enumerate}

\subsection*{Erinnerung -- Taylor-Entwicklung in $\mathbb{R}$}
\begin{enumerate}
    \item Sei $f:(a,b)\to\mathbb{R}$ $(r+1)$-mal stetig differenzierbar. Dann gilt fĂĽr $x,x+h\in(a,b)$:
    $$
    f(x+h)=\sum_{k=0}^{r}\frac{f^{(k)}(x)}{k!}\,h^k+R_{r+1}(x,h).
    $$

    \item FĂĽr das Restglied in Lagrange-Form (fĂĽr $0\in(0,1)$ bezeichnet hier typischerweise $\theta\in(0,1)$) gilt:
    $$
    R_{r+1}(x,h)=\frac{f^{(r+1)}(x+\theta h)}{(r+1)!}\,h^{\,r+1}
    $$
    fĂĽr ein geeignetes $\theta\in(0,1)$.
    \item FĂĽr das Restglied in Integral-Form gilt:
    $$
    R_{r+1}(x,h)=\frac{h^{r+1}}{r!}\int_0^1 (1-t)^r\,f^{(r+1)}(x+th)\,dt.
    $$
\end{enumerate}

\begin{satz}{Taylor-Formel}
    Seien $D\subset\mathbb{R}^n$ offen, $x\in D$, $h\in\mathbb{R}^n$ mit $\{x+th\mid t\in[0,1]\}\subset D$ und $f\in C^{r+1}(D,\mathbb{R})$.

    Dann existiert ein $\theta\in[0,1]$, sodass gilt:
    $$
    \begin{aligned}
        f(x+h) & =f(x)
        +\underbrace{\sum_{k=1}^r\frac{1}{k!}\sum_{i_1,\dots,i_k=1}^n
        \frac{\partial^k f}{\partial x_{i_k}\dots\partial x_{i_1}}(x)\,
        h_{i_1}\cdots h_{i_k}}_{\text{Taylor-Polynom}} \\
        & +\underbrace{\frac{1}{(r+1)!}\sum_{i_1,\dots,i_{r+1}=1}^n
        \frac{\partial^{r+1}f}{\partial x_{i_{r+1}}\dots\partial x_{i_1}}(x+\theta h)\,
        h_{i_1}\cdots h_{i_{r+1}}}_{\text{Restglied}}.
    \end{aligned}
    $$
        
\end{satz}

\begin{korollar}{Restglied--Taylor--Formel}
    Seien $D\subset\mathbb{R}^n$ offen, $f\in C^{r+1}(D,\mathbb{R})$.

    Dann gilt fĂĽr alle $x\in D$ und $h\in\mathbb{R}^n$ mit $x+th\in D$ ($\forall t\in[0,1]$):
    $$
    f(x+h)=\sum_{|\alpha|\le r+1}\frac{\partial^\alpha f(x)}{\alpha!}\,h^\alpha+\omega_{r+1}(x,h),
    $$
    wobei
    $$
    \frac{\omega_{r+1}(x,h)}{\|h\|^{r+1}}\longrightarrow 0\quad\text{fĂĽr }h\to 0,
    $$
    das heiĂźt\ $\omega_{r+1}(x,h)=o(\|h\|^{r+1})$.
\end{korollar}

\begin{korollar}{Taylor--Formel 1. und 2. Ordnung}
    Seien $D\subset\mathbb{R}^n$ offen, $f\in C^1(D,\mathbb{R})$.
    Dann gilt fĂĽr alle $x\in D$ und $h\in\mathbb{R}^n$ mit $x+th\in D$ fĂĽr alle $t\in[0,1]$:
    $$
    f(x+h)=f(x)+(\nabla f(x),h)_2+\omega_1(x,h),
    \qquad\text{mit}\qquad
    \frac{\omega_1(x,h)}{\|h\|}\longrightarrow 0\quad (h\to 0).
    $$
    D.\,h. das Taylor-Polynom 1. Ordnung ist eine lineare Approximation von $f$.

    FĂĽr $f\in C^2(D,\mathbb{R})$ gilt:
    $$
    f(x+h)=f(x)+(\nabla f(x),h)_2+\tfrac{1}{2}\big(H_f(x)h,h\big)_2+\omega_2(x,h),
    \qquad\text{mit}\qquad
    \frac{\omega_2(x,h)}{\|h\|^2}\longrightarrow 0\quad (h\to 0),
    $$
    wobei $H_f(x)$ die Hesse-Matrix von $f$ ist.

    D.\,h. das Taylor-Polynom 2. Ordnung ist eine quadratische Approximation von $f$.
\end{korollar}

\begin{definition}{Taylor--Reihe}
    Seien $D\subset\mathbb{R}^n$ offen und $f: D\to\mathbb{R}$ beliebig oft partiell differenzierbar, $x\in D$.

    Dann heiĂźt die Taylor-Reihe von $f$ in $x$:
    $$
    T^{f}_{\infty}(x+h)=\sum_{|\alpha|=0}^{\infty}\frac{\partial^{\alpha}f(x)}{\alpha!}\,h^{\alpha}.
    $$
\end{definition}

\begin{korollar}{Konvergenz von Taylor--Reihen}
    Seien $D\subset\mathbb{R}^n$ offen und $f: D\to\mathbb{R}$ eine beliebig oft differenzierbare Funktion.

    Dann gilt
    $$
    T_r^f(x+h)=\sum_{|\alpha|=0}^r \frac{\partial^\alpha f(x)}{\alpha!}\,h^\alpha
    \xrightarrow{r\to\infty} f(x+h),
    $$
    wenn
    $$
    R_{r+1}^f(x,h)\xrightarrow{r\to\infty}0,\qquad x\in D. (**)
    $$
    Eine hinreichende Bedingung fĂĽr (**) ist
    $$
    \sup_{|\alpha|\ge 0}\sup_{x\in D} \bigl|\partial^\alpha f(x)\bigr| \le M_f < \infty,
    $$
    das heiĂźt, alle partiellen Ableitungen von $f$ sind gleichmĂ¤Ăźig beschrĂ¤nkt.
\end{korollar}
