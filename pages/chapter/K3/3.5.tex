\section{Banachsche Fixpunktsatz}
\label{3.5}

\begin{definition}{Fixpunkt}
     Sei $g : X \to X$ eine Abbildung. Ein Punkt $a \in X$ ist ein \emph{Fixpunkt} der Abbildung $g$ wenn $g(a) = a$ gilt.   
\end{definition}

\begin{definition}{Kontraktion}
    Sei $(X, d)$ ein metrischer Raum. Eine Abbildung $g : X \to X$ heißt \emph{Kontraktion}, falls es ein $q  \in (0, 1)$ gibt mit
    $$
        d(g(x), g(y)) \le q \cdot d(x, y), \forall x, y \in X
    $$
\end{definition}

\begin{satz}{Banachscher Fixpunktsatz}
    COMMING SOON
\end{satz}

\begin{definition}{Starke Monotonie}
    Eine Funktion $f : D \subset \R^n \to \R^n$ heißt stark monoton, wenn eine Konstante $m > 0$ existiert, sodass $\forall x, y \in D$ gilt
    $$
        (f(x) - f(y), x - y)_2 \ge m \|x - y\|_2^2.
    $$
\end{definition}
