\section{Banachsche Fixpunktsatz}
\label{3.5}

\begin{bemerkungbox}
    \textbf{Fixpunktsatz:} Ein einfaches, aber sehr wirksames Mittel zum Beweis vieler Existenzsätze.

    \textbf{Motivation:} Es sei ein quadratisches Gleichungssystem der Form
    $$
    \begin{aligned}
    f_1(x_1,\dots,x_n) &= b_1,\\
    &\vdots\\
    f_n(x_1,\dots,x_n) &= b_n,
    \end{aligned}
    \qquad\Longleftrightarrow\qquad f(x)=b.
    $$

    gegeben, wobei $x\in K^n$,
    $$
    f=\begin{pmatrix} f_1\\ \vdots\\ f_n\end{pmatrix}:\; E\subset K^n\to K^n,\qquad
    b=\begin{pmatrix} b_1\\ \vdots\\ b_n\end{pmatrix}\in K^n.
    $$

    Ziel: $x=f^{-1}(b)$ finden als Grenzwert einer Folge $(x^{(k)})_{k\in\mathbb{N}}$, d.\,h.\ $x$ so dass $f(x)=b$.

    \medskip

    \textbf{Ansatz:} Definiere
    $$
    g(x):=x-\sigma\bigl(f(x)-b\bigr)
    $$
    für ein $\sigma\in K\setminus\{0\}$ und suche einen Fixpunkt von $g:D\to K^n$ (also $x=g(x)$).

    \textbf{Fixpunktiteration:}
    \begin{itemize}
    \item Startwert $x^{(0)}$.
    \item Iterationsschritt $x^{(k)}=g\bigl(x^{(k-1)}\bigr),\qquad k\in\mathbb{N}.$
    \end{itemize}

    Falls $f$ stetig ist, dann ist auch $g$ stetig. Damit folgt:

    Ist $x^{(k)}\to x$ für $k\to\infty$, so gilt wegen der Stetigkeit von $g$
    $$
    g\bigl(x^{(k-1)}\bigr)\to g(x).
    $$
    Da $x^{(k)}=g\bigl(x^{(k-1)}\bigr)$, folgt im Grenzwert $x=g(x)$, also ist $x$ ein Fixpunkt von $g$. Aus
    $$
    x=g(x)=x-\sigma\bigl(f(x)-b\bigr)
    $$
    folgt $\sigma\bigl(f(x)-b\bigr)=0$ und damit $f(x)=b$.

    \medskip

    \textbf{Frage:} Unter welchen Bedingungen konvergiert die Fixpunktiteration?
\end{bemerkungbox}

\begin{definition}{Fixpunkt}
     Sei $g : X \to X$ eine Abbildung. Ein Punkt $a \in X$ ist ein \emph{Fixpunkt} der Abbildung $g$ wenn $g(a) = a$ gilt.   
\end{definition}

\begin{definition}{Kontraktion}
    Sei $(X, d)$ ein metrischer Raum. Eine Abbildung $g : X \to X$ heißt \emph{Kontraktion}, falls es ein $q  \in (0, 1)$ gibt mit
    $$
        d(g(x), g(y)) \le q \cdot d(x, y), \forall x, y \in X
    $$
\end{definition}

\begin{satz}{Banachscher Fixpunktsatz}
    Sei
    \begin{itemize}
    \item $(X,d)$ ein vollständiger metrischer Raum,
    \item $D\subset X$ abgeschlossen,
    \item $g:D\to D$ eine Kontraktion.
    \end{itemize}

    Dann gilt:
    \begin{enumerate}%[label=(\roman*)]
    \item $g$ besitzt genau einen Fixpunkt $a\in D$.
    \item Ist $x^{(0)}\in D$ ein beliebiger Startwert, so konvergiert die Folge der Iterationen
    $$
    x^{(0)},\; x^{(1)}=g\bigl(x^{(0)}\bigr),\; \ldots,\; x^{(k+1)}=g\bigl(x^{(k)}\bigr),\; \ldots
    $$
    gegen den Fixpunkt $a$.
    \item Es gilt die Fehlerabschätzung:
    $$
    d\bigl(a,x^{(k)}\bigr)\le \frac{1}{1-q}\,d\bigl(x^{(k+1)},x^{(k)}\bigr)
    \le \frac{q^{k}}{1-q}\,d\bigl(x^{(1)},x^{(0)}\bigr).
    $$
    \end{enumerate}
\end{satz}

\begin{definition}{Starke Monotonie}
    Eine Funktion $f : D \subset \R^n \to \R^n$ heißt stark monoton, wenn eine Konstante $m > 0$ existiert, sodass $\forall x, y \in D$ gilt
    $$
        (f(x) - f(y), x - y)_2 \ge m \|x - y\|_2^2.
    $$
\end{definition}

\begin{satz}{Nichtlineare Gleichungssysteme}
    Sei $f:\mathbb{R}^n\to\mathbb{R}^n$ eine Lipschitz-stetige, stark monotone Abbildung mit Lipschitzkonstante $L>0$ und Monotoniekonstante $m>0$.

    Dann hat die Gleichung
    $$
    f(x)=b
    $$
    für jedes beliebige $b\in\mathbb{R}^n$ eine eindeutige Lösung $x^*\in\mathbb{R}^n$.

    Für jeden Startpunkt $x^{(0)}\in\mathbb{R}^n$ konvergiert die Iteration
    $$
    x^{(k+1)}=x^{(k)}-\sigma\bigl(f(x^{(k)})-b\bigr)
    $$
    für jedes $\sigma\in\left]0,\frac{2m}{L^2}\right[$ gegen $x^*$.   
\end{satz}
