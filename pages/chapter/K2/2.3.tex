\section{Kompaktheit}
\label{2.3}
\begin{definition}[deflabel=Kompaktheit]{Kompaktheit}
    Sei $(X,d)$ ein metrischer Raum und eine Menge $K\subset X$ gegeben.

    Eine \emph{offene Überdenkung} von $K$ ist eine Familie $\{V_i\_{i\in I}\}$ von offenen Mengen $V_i \subset X$ mit der Indexmenge $I$ und $K\subset \bigcup_{i \in I} V_i$. Dann heißt $K$
    \begin{enumerate}
        \item[(i)] \emph{folgenkompakt}, wenn für jede Folge $(x_k)_{k\in\N}$ in $K$ eine konvergente Teilfolge $(x_{k_j})_{k,j \in \N}$ und ein Grenzwert $x \in K$ existiert mit $\lim_{j \to \infty} x_{k_j}$.
        \item[(ii)] \emph{Überdeckungskompakt} oder auch nur \emph{kompakt}, wenn es zu jederoffenen Überdekcung von $K$ eine endliche Teilüberdeckung git, das heißt:
        Für jede Familie offener Mengen $\{V_i\}_{i\in I}$ mit $K \subset \bigcup_{i \in I} V_i$ gibt es endlich viele Indizes
        $$
            M = \{i_1,..., i_n\} \subset I, \quad K \subset \bigcup_{i \in I} V_i
        $$
    \end{enumerate}
\end{definition}

\begin{satz}{metrische Räume: Überdeckungskompaktheit $\Leftrightarrow$ Folgenkompaktheit}
    Sei $(X,d)$ ein metrischer Raum und $K\subset X$.

    \vspace{1em}

    Dann ist $K$ (überdeckungs-)kompakt genau dann, wenn $K$ folgenkompakt ist.
\end{satz}

\begin{lemma}{Hilfslemma:}
    COMMING SOON
\end{lemma}

\begin{satz}{metrische Räume: Kompakt $\Rightarrow$ abgeschlossen und beschränkt}
    In einem metrischen Raum ist jede kompakte Menge abgeschlossen und beschränkt.
\end{satz}

\begin{satz}{}
    Sei $(X,d)$ ein metrischer Raum, $K\subset X$ kompakt und $A\subset K$ abgeschlossen. Dann ist $A$ kompakt.
\end{satz}

\begin{satz}{Heine--Borel}
    Sei $(\R^n, d)$ der metrische Raum aller reelen $n$-Tupel mit der euklidischen Metik.

    Dann sind folgende Aussagen äquivalent:
    \begin{enumerate}
        \item[(i)] $K$ ist kompakt.
        \item[(ii)] $K$ ist beschränkt und abgeschlossen.
    \end{enumerate}
\end{satz}

\begin{satz}{Hilfssatz}
    Es seien $a_i,b_i\in\mathbb{R},\ a_i\le b_i,\ i=1,\dots,n.$

    Dann ist der abgeschlossene Quader
    $$
    Q:=\left\{\begin{pmatrix} x_1\\ \vdots\\ x_n\end{pmatrix}\in\mathbb{R}^n:\ a_i\le x_i\le b_i,\ i=1,\dots,n\right\}
    $$
    kompakt.
\end{satz}
