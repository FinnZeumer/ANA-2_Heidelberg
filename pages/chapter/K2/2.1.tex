\section{Normierte und metrische Räume, der Raum $\mathbb{R}^n$}
\label{sec:2.2}

Kurze Bemerkung zu beginn:
Der Raum $\R^n$, $n \in \N$, besteht aus allen $n$-Tupeln (oder \emph{Vektoren}):

$$
\mathbb{R}^n = \left\{ x = \begin{pmatrix} x_1 \\ \vdots \\ x_n \end{pmatrix} : x_i \in \mathbb{R} \text{ für alle } i = 1, \dots, n \right\} \, .
$$

Für $x = \begin{pmatrix} x_1 \\ \vdots \\ x_n \end{pmatrix}, \quad y = \begin{pmatrix} y_1 \\ \vdots \\ y_n \end{pmatrix} \in \mathbb{R}^n$ und \emph{Skalare} $\alpha \in \R$ defineiren wir auf $\R^n$ \emph{Vektoraddition} und \emph{skalare Multiplikation} durch

$$
x + y := \begin{pmatrix} x_1 + y_1 \\ : \\ \vdots \\ x_n + y_n \end{pmatrix}, \quad \alpha \cdot x := \begin{pmatrix} \alpha x_1 \\ : \\ \vdots \\ \alpha x_n \end{pmatrix} \, .
$$

\begin{definition}{Metrik, metrischer Raum}
    Sei $X$ eine Menge. Eine Abbildung 
    $$
    d : X \times X \to \mathbb{R}, \qquad (x,y) \to d(x,y)
    $$
    heißt \emph{Metrik}, wenn für alle $x,y,z \in X$ gilt:

    \begin{enumerate}
        \item[(M1)] Positiv Definitheit: $d(x,y)\geq 0$ und $d(x,y)=0 \Leftrightarrow x=y$,
        \item[(M2)] Symmetrie: $d(x,y)=d(y,x)$,
        \item[(M3)] Dreiecksungleichung: $d(x,z) \le d(x,y) + d(y,z)$.
    \end{enumerate}

    Ein metrischer Raum ist ein Paar $(X,d)$, bestehend aus einer Menge $X$ und einer Metrik $d$.
    Man nennt $d(x, y)$ auch den Abstand oder die Distanz von $x$ und $y$.
\end{definition}

\begin{beispielbox}
    \begin{enumerate}
        \item Sei \(X=\mathbb{R}\) oder \(\mathbb{C}\) mit
        \[
        d(x,y):=|x-y|.
        \]
        Dann ist \((X,d)\) ein metrischer Raum.

        \medskip

        Begründung der Axiome:
        \begin{itemize}
        \item[M1] Folgt aus \(|x|=0 \Leftrightarrow x=0\).
        \item[M2] Folgt aus \(|x-y|=|y-x|\) (Symmetrie).
        \item[M3] Folgt aus der Dreiecksungleichung für den Betrag:
        \[
        d(x,z)=|x-z|=|x-y+y-z|\le |x-y|+|y-z|=d(x,y)+d(y,z).
        \]
        \end{itemize}

        Beobachtung:
        Aus M1--M3 folgt \(d(x,y)\ge 0\) für alle \(x,y\in X\). Tatsächlich ist
        \[
        0=d(x,x)\stackrel{\text{M3}}{\le} d(x,y)+d(y,x)\stackrel{\text{M2}}{=}2\,d(x,y),
        \]
        also \(d(x,y)\ge0\).

        \item (induzierte Metrik) Sei \((X,d)\) ein metrischer Raum und \(A\subset X\). Dann definiert
        \[
        d_A : A\times A\to\mathbb{R},\qquad d_A(x,y):=d(x,y)
        \]
        eine Metrik auf \(A\).

        \item (triviale Metrik) Auf jeder Menge \(X\) kann man die triviale Metrik definieren durch
        \[
        d(x,y):=\begin{cases}
        0, &\text{falls }x=y,\\[4pt]
        1, &\text{falls }x\ne y.
        \end{cases}
        \]
    \end{enumerate}

    \medskip

    Weitere wichtige Beispiele: metrische Räume entstehen aus normierten Vektorräumen.
\end{beispielbox}

\begin{definition}[deflabel=NormRaum]{Norm, normierter Raum}
    Sei $V$ ein Vektorraum über einen $\K \, (\mathbb{R}$ oder $\mathbb{C}$). Eine Abbildung 
    $$
    \|\cdot\| : V \to \mathbb{R}$$
    heißt \emph{Norm} auf $V$, wenn für alle $x,y \in V$ und $\alpha \in \K$ gilt:

    \begin{enumerate}
        \item[(N1)] Definitheit: $\|x\| = 0 \Leftrightarrow x=0$,
        \item[(N2)] Homogenität: $\|\alpha x\| = |\alpha|\cdot\|x\|$,
        \item[(N3)] Dreiecksungleichung: $\|x+y\| \le \|x\| + \|y\|$.
    \end{enumerate}

    Das Paar $(V,\|\cdot\|)$ heißt normierter Raum.
\end{definition}

\begin{bemerkungbox}
    Aus den Eigenschaften N1 bis N3 folgt:
    $$
        \left\|x\right\| \geq 0, \quad \forall x \in V
    $$
\end{bemerkungbox}

\begin{beispielbox}
    \begin{align*}
        \|x\|_1 &= \sum_{i=1}^n |x_i| && \text{(Summennorm / Manhattan-Norm)} \\
        \|x\|_2 &= \sqrt{\sum_{i=1}^n x_i^2} && \text{(Euklidische Norm)} \\
        \|x\|_\infty &= \max_{i=1,\dots,n} |x_i| && \text{(Maximumsnorm / Schachbrett-Norm)}
    \end{align*}

    \textbf{Weitere Beispiele:}
    \begin{enumerate}
        \item \textbf{Triviale Metrik:} Für eine beliebige Menge $M$ sei
        $$
        d(x, y) := \begin{cases}
            0 & \text{wenn } x = y \\
            1 & \text{wenn } x \neq y
        \end{cases}
        $$

        \item \textbf{Induzierte Metrik:} Sei $(M,d)$ ein metrischer Raum und $N \subset M$. Dann ist die auf $N$ \emph{induzierte Metrik} gegeben durch:
        $$
            d_N(x, y) := d(x, y), \quad \forall x, y \in N
        $$

        \item \textbf{Supremumsnorm:} Sei
        $$
        \mathcal{B}(X) := \{ f : X \to \mathbb{R} \mid f \text{ beschränkt} \}
        $$
        Dann ist $\|\cdot\|_\infty := \sup_{x \in X} |f(x)|$ eine Norm auf $\mathcal{B}(X)$ mit:
        \begin{enumerate}
            \item[(N1)] $\|f\|_\infty = 0 \Lra f(x) = 0 \quad \forall x \in X \quad \color{steelblue}\surd$
            \item[(N2)] $\| \lambda f \|_\infty = |\lambda| \cdot \|f\|_\infty \quad \color{steelblue}\surd$
            \item[(N3)] 
                \begin{align*}
                    \|f+g\|_\infty &= \sup_{x \in X} |f(x) + g(x)| \\
                    &\le \sup_{x \in X} \big(|f(x)| + |g(x)|\big) \\
                    &\le \|f\|_\infty + \|g\|_\infty \quad \color{steelblue}\surd
                \end{align*}

        \item \textbf{$\ell_p$-Nornem}
        \img[0.4]{K2/VerschiedeneNormen}{Visualisierung der Entwicklung der $\ell_p$-Normen.}
        \end{enumerate}
    \end{enumerate}
\end{beispielbox}


\begin{lemma}{Metrisierung normierter Räume}
    Sei $(V,\|\cdot\|)$ ein normierter Raum. Dann in $V$, versehen mit der Metrik $d : V \times V \rightarrow \mathbb{R}$ definiert durch
    $$
        d(x,y) := \|x-y\| \,
    $$
    eine \emph{metrischer Raum}.
\end{lemma}

\begin{beweisbox}
    Z.\,z. \(d\) ist eine Metrik, d.\,h. \(d\) erfüllt M1--M3.

    Sei \(d(x,y)=\lVert x-y\rVert\) für einen normierten Raum.

    \begin{itemize}
    \item M1: \(\) Aus N1 folgt
        \[
        \lVert x-y\rVert = 0 \iff x=y,
        \]
        also ist M1 erfüllt. \(\checkmark\)

    \item M2: 
        \[
        d(x,y)=\lVert x-y\rVert \overset{\text{N2}}{=} \lVert y-x\rVert = d(y,x),
        \]
        da \(\lVert(-1)(x-y)\rVert=\lVert x-y\rVert\). Somit ist M2 erfüllt. \(\checkmark\)

    \item M3 (Dreiecksungleichung):
        Für beliebige \(x,y,z\) gilt
        \[
        d(x,y)=\lVert x-y\rVert = \lVert x-z + z-y\rVert \le \lVert x-z\rVert + \lVert z-y\rVert
        = d(x,z)+d(z,y),
        \]
        also ist M3 erfüllt.
    \end{itemize}

    Beispiel in \(\mathbb{R}^n\): Sei \(\lVert\cdot\rVert_2\) die \(\ell_2\)-Norm. Dann
    \[
    d_2(x,y)=\sqrt{\sum_{i=1}^n (x_i-y_i)^2},
    \]
    die euklidische Metrik.
\end{beweisbox}

\begin{lemma}{Umgekehrte Dreiecksungleichung}
    Sei $(X,d)$ ein metrischer Raum und $x,y,z \in X$. Dann gilt:
    $$
        |d(x,z) - d(y,z)| \le d(x,y).
    $$
\end{lemma}

\begin{beweisbox}
    Aus der Dreiecksungleichung gilt
    \begin{align*}
    d(y,z) &\le d(y,x)+d(x,z) \\
        &= d(x,y)+d(x,z) \quad(\text{Symmetrie})\\
    \Rightarrow\quad -\,d(x,y)&\le d(x,z)-d(y,z).
    \end{align*}
    Weiter gilt ebenfalls
    \begin{align*}
    d(x,z) &\le d(x,y)+d(y,z) \\
    \Rightarrow\quad d(x,z)-d(y,z)&\le d(x,y).
    \end{align*}
    Kombiniert erhält man
    \[
    -\,d(x,y)\le d(x,z)-d(y,z)\le d(x,y),
    \]
    also
    \[
    \lvert d(x,z) - d(y,z)\rvert \le d(x,y).
    \text{In der Vorlesung: }\lvert d(x,z) \textcolor{red}{+} d(y,z)\rvert \le d(x,y).
    \]
    q.e.d.
\end{beweisbox}

\begin{definition}{Offene und abgeschlossene Kugeln}
    Sei $(X,d)$ ein metrischer Raum. Für $x \in X$ und $R>0$ definiert man
    \begin{enumerate}
        \item die \emph{offene Kugel} $B_R(x)$ um $x$ und Raidus $R$
        $$
            B_R(x) := \{y \in X : d(x,y) < R\} \, .
        $$

        \item die \emph{abgeschlossene Kugel} $\overline{B}_R(x)$ um $x$ und Raidus $R$
        $$
        \overline{B}_R(x) := \{y \in X : d(x,y) \le R\}.
        $$
    \end{enumerate}
\end{definition}

\begin{definition}{Beschränkte Menge}
    Sei $(X,d)$ ein metrischer Raum. Eine Menge $E \subset X$ heißt \emph{beschränkt}, wenn es ein $x \in X$ und $R>0$ gibt mit
    $$
        E \subset B_R(x).
    $$
\end{definition}

Nun wo die wichtigsten Definitonen geklärt sind, soll nun die \emph{ Konvergenz in metrischen Räumen} betrachtet werden.

\begin{definition}{Konvergenz von Folgen}
    Sei $(X,d)$ ein metrischer Raum und $(x_k)_{k\in\N}$ eine Folge in $X$. Die Folge \emph{konvergiert} gegen $a \in X$, wenn gilt:
    $$
        \forall \varepsilon>0, \quad \exists k_0 \in \N:\; d(x_k,a) < \varepsilon \quad \forall k \ge k_0.
    $$

    In diesem Fall heißt $a$ \emph{Limes} oder \emph{Grenzwert} der Folge und man schreibt 
    $$
        \alpha = \lim\limits_{k \to \infty} x_k \quad \text{oder} \quad x_k \to a \text{ für } k \to \infty
    $$

    Eine Folge, welche konvergiert, heißt \emph{konvergent}. Eine Folge, die nicht konvergiert, heißt \emph{divergent}.
\end{definition}

\begin{bemerkungbox}
    \textbf{Beobachtungen}
    \begin{enumerate}
        \item Der Grenzwert der Folge ist eindeutig.

            Seien \(a,\widetilde{a}\in X\) mit
            \[
            \lim_{k\to\infty} x_k = a,\qquad \lim_{k\to\infty} x_k = \widetilde{a}.
            \]
            Sei \(\varepsilon>0\). Dann gibt es \(k_0,k_1\in\mathbb{N}\) mit
            \[
            d(x_k,a)<\varepsilon\quad\text{für alle }k\ge k_0,
            \qquad
            d(x_k,\widetilde{a})<\varepsilon\quad\text{für alle }k\ge k_1.
            \]
            Setze \(k_2=\max\{k_0,k_1\}\). Für alle \(k\ge k_2\) gilt dann
            \[
            d(a,\widetilde{a}) \le d(a,x_k)+d(x_k,\widetilde{a}) < \varepsilon+\varepsilon = 2\varepsilon.
            \]
            Da \(\varepsilon>0\) beliebig war, folgt \(d(a,\widetilde{a})=0\) und somit \(a=\widetilde{a}\).

        \item 
            \(
            \lim_{k\to\infty} x_k = a \quad\Longleftrightarrow\quad \text{In jeder }\varepsilon\text{-Kugelumgebung }B_\varepsilon(a)\text{ liegen fast alle }x_k.
            \)
            \noindent
            d.\,h.\ alle bis auf endlich viele.

            \begin{beweisbox}
            "$\Rightarrow$": Sei $\varepsilon>0$. Da $x_k\to a$ gilt, existiert ein Index $k_\varepsilon\in\mathbb{N}$ mit
            \[
            \forall k\ge k_\varepsilon:\; x_k\in B_\varepsilon(a).
            \]
            Somit ist die Menge
            \[
            \{\,k\in\mathbb{N}:\; x_k\notin B_\varepsilon(a)\,\}
            \]
            eine Teilmenge von $\{1,2,\dots,k_\varepsilon-1\}$ und daher endlich. Insbesondere gilt
            \[
            \#\{k:\; x_k\notin B_\varepsilon(a)\} < k_\varepsilon <\infty.
            \]

            "$\Leftarrow$": Sei nun vorausgesetzt, dass für jedes $\varepsilon>0$ die Menge
            \[
            S_\varepsilon:=\{\,k\in\mathbb{N}:\; x_k\notin B_\varepsilon(a)\,\}
            \]
            endlich ist. Dann besitzt $S_\varepsilon$ ein Maximum $k_\varepsilon$ (wobei man im Fall $S_\varepsilon=\varnothing$ z.\,B. $k_\varepsilon:=0$ setzen kann). Für alle $k>k_\varepsilon$ gilt daher $x_k\in B_\varepsilon(a)$, also $d(x_k,a)<\varepsilon$. Da dies für beliebiges $\varepsilon>0$ gilt, folgt $x_k\to a$.
            q.e.d.
            \end{beweisbox}

        \item
            \(
            \lim_{k\to\infty} x_k = a \iff \lim_{k\to\infty} d(x_k,a) = 0.
            \)

            In anderen Worten: \((d_k)_{k\in\mathbb{N}}\) ist eine Nullfolge in \(\mathbb{R}\),

            \[
            d_k = d(x_k,a) \in \mathbb{R}.
            \]  
    \end{enumerate}
\end{bemerkungbox}

\begin{definition}{Häufungspunkt}
    Sei $(X,d)$ ein metrischer Raum und $(x_k)_{k\in\N}$ eine Folge in $X$. 

    Ein Punkt $a \in X$ heißt \emph{Häufungspunkt} der Folge, wenn eine Teilfolge $(x_{k_j})_{k,j\in\N}$ existiert, welche gegen $a$ konvergiert.
\end{definition}

\begin{beispielbox}
    $$
        \text{im } \R \quad x_k = (-1)^k = \{-1,1,-1,1,\dots\}
    $$
\end{beispielbox}

\begin{bemerkungbox}
    \textbf{Beobachtungen:}
    \begin{itemize}
        \item Der Grenzwert einer Folge ist ein Häufungspunkt der Folge. Jede konvergente Folge besitzt genau einen Häufungspunkt.
        \item Sei $(x_k)_{k\in\mathbb{N}}$ eine Folge in einem metrischen Raum $X$. Dann ist $a\in X$ genau dann ein Häufungspunkt der Folge, wenn in jeder $\varepsilon$-Kugelumgebung von $a$ unendlich viele $x_k$ liegen.
    \end{itemize}
\end{bemerkungbox}

\begin{definition}{Konvergenz in $\mathbb{R}^n$}
    Sei $(x_k)_{k\in\N}$ eine Folge in $\R^n$,
     $$
     x_k =
        \begin{pmatrix}
        x_{k,1}\\
        \vdots \\
        x_{k,n}
        \end{pmatrix} \, ,
    \quad
    a =
        \begin{pmatrix}
        a_1 \\
        \vdots \\
        a_n
        \end{pmatrix}
        \in \mathbb{R}^n \, .
    $$ 

    Dann heißt die folge \emph{konvergent} gegen ein $a \in \R^n$, wenn
    $$
        \|x_k - a\|_\infty \to 0,\qquad k \to \infty \, .
    $$
\end{definition}

\begin{bemerkungbox}
    Sei \((x_k)_{k\in\mathbb{N}}\) eine Folge in \(\mathbb{R}^n\) und \(a\in\mathbb{R}^n\). Dann gilt
    \[
    \|x_k-a\|_{\infty}\to 0,\qquad k\to\infty
    \quad\Longleftrightarrow\quad
    \forall i=1,\dots,n:\ \lim_{k\to\infty} x_{k,i}=a_i.
    \]

    Komponentenweise Konvergenz in \(\mathbb{R}^n\).

    Insbesondere
    \[
    \|x_k-a\|_{\infty}=\max_{i=1,\dots,n}\big|x_{k,i}-a_i\big|
    \xrightarrow[k\to\infty]{}0,
    \]
    also für jedes \(i\)
    \[
    \big|x_{k,i}-a_i\big|\xrightarrow[k\to\infty]{}0.
    \]
\end{bemerkungbox}

\begin{satz}{Bolzano-Weierstraß}
    Jede beschränkte Folge in $\mathbb{R}^n$ besitzt eine konvergente Teilfolge.
\end{satz}

\begin{beweisbox}
    Sei \((x_k)_{k\in\mathbb{N}}\) beschränkt in \(\mathbb{R}^n\).
    Dann sind für alle \(i=1,\dots,n\) die Komponentenfolgen \((x_{k,i})_{k\in\mathbb{N}}\) beschränkt in \(\mathbb{R}\).

    Nach dem Bolzano--Weierstraßsatz in \(\mathbb{R}\) existiert eine konvergente Teilfolge
    \((x_{k_j^{(1)},1})_{j\in\mathbb{N}}\) der ersten Komponentenfolge \((x_{k,1})_{k\in\mathbb{N}}\)
    mit
    \[
    x_{k_j^{(1)}}\longrightarrow a_1\in\mathbb{R}\qquad (j\to\infty).
    \]

    Aus der so erhaltenen Teilfolge betrachten wir die zweite Komponente.
    Da auch diese beschränkt ist, existiert eine Teilfolge
    \((x_{k_j^{(2)},2})_{j\in\mathbb{N}}\) von \((x_{k_j^{(1)},2})_{j\in\mathbb{N}}\)
    mit
    \[
    x_{k_j^{(2)},2}\longrightarrow a_2\in\mathbb{R}\qquad (j\to\infty).
    \]

    Induktiv verfahren wir für \(i=3,\dots,n\): Aus der bereits gewählten Teilfolge
    wählen wir jeweils eine weitere Teilfolge, so dass die \(i\)-te Komponente konvergiert.
    Dadurch erhalten wir verschachtelte Indizes
    \[
    k_1^{(1)}<k_2^{(1)}<\dots,\qquad
    k_1^{(2)}<k_2^{(2)}<\dots,\qquad\ldots,\qquad
    k_1^{(n)}<k_2^{(n)}<\dots
    \]
    mit \(k_j^{(i+1)}\in\{k_\ell^{(i)}:\ell\in\mathbb{N}\}\) für alle \(i\).

    Betrachte die Diagonalfolge \(k_j:=k_j^{(n)}\). Dann konvergiert für jede Komponente
    \[
    x_{k_j,i}\longrightarrow a_i\in\mathbb{R}\qquad (j\to\infty),\quad i=1,\dots,n \, .
    \]

    Ergebnis: Es existiert eine konvergente Teilfolge \((x_{k_j})_{j\in\mathbb{N}}\) von
    \((x_k)_{k\in\mathbb{N}}\) mit \(x_{k_j}\to a\). \(\square\)
\end{beweisbox}

\begin{satz}{Äquivalenz der Normen}
    In einem endlichdimensionalen Vektorraum. $\mathbb{R}^n$ sind alle Normen äquivalent zur Maximumnorm $(l_\infty)$.D.h. zu jeder Norm $\|\cdot\|$ existieren Konstanten $m,M>0$, sodass
    $$
        m\|x\|_\infty \le \|x\| \le M\|x\|_\infty \quad \text{für alle } x \in \mathbb{R}^n.
    $$
\end{satz}

\subsection*{Cauchy-Folgen und Vollständigkeit metrischer Räume}

\begin{definition}{Cauchy-Folge}
    Sei $(X,d)$ ein metrischer Raum. Eine Folge $(x_k)_{k\in\N}$ in $X$ heißt \emph{Cauchy-Folge}, falls für alle $\varepsilon>0$ ein $k^*\in\mathbb{N}$ existiert, sodass
    $$
        d(x_k,x_m) < \varepsilon \quad \forall k,m \ge k^*.
    $$
\end{definition}

\begin{bemerkungbox}
    $$
        \R \qquad (x_k)_{k\in\N} \subset \R \text{ ist eine C.F., falls}
    $$
    $$
        \forall \epsilon > 0 \exists K^* \text{, sodass } |x_k - x_m| < \epsilon
    $$
    $$
        \forall k,m \geq k^*
    $$
\end{bemerkungbox}

\begin{satz}{Eigenschaften von Cauchy-Folgen}
    In einem metrischen Raum gilt:
    \begin{itemize}
        \item[(i)] Jede konvergente Folge ist eine Cauchy-Folge.
        \item[(ii)] Jede Cauchy-Folge ist beschränkt.
        \item[(iii)] Eine Cauchy-Folge hat höchstens einen Häufungspunkt. Sie konvergiert genau dann, wenn sie einen Häufungspunkt besitzt.
    \end{itemize}
\end{satz}

\begin{beweisbox}
    \begin{enumerate}
    \item[(i)] Sei \((x_k)_{k\in\mathbb{N}}\) und \(x_k\to x\).

    Sei \(\varepsilon>0\) gegeben. Dann existiert \(k^*\in\mathbb{N}\) so, dass
    \[
    d(x_k,x)<\frac{\varepsilon}{2}\qquad\text{für alle }k\ge k^*
    \]
    (Definition des Grenzwerts). Für alle \(k,n\ge k^*\) gilt dann wegen der Dreiecksungleichung
    \[
    d(x_k,x_n)\le d(x_k,x)+d(x,x_n)<\frac{\varepsilon}{2}+\frac{\varepsilon}{2}=\varepsilon.
    \]
    Damit ist \((x_k)\) eine Cauchy-Folge.

    \item[(ii)] Sei \((x_k)_{k\in\mathbb{N}}\) eine Cauchy-Folge. Setze \(\varepsilon=1\). Dann existiert \(k^*\) so, dass
    \[
    d(x_{k^*},x_n)<1\qquad\text{für alle }n\ge k^*.
    \]
    Definiere
    \[
    R:=1+\max\{\,d(x_\ell,x_{k^*}):\ell=1,\dots,k^*-1\,\}<\infty.
    \]
    Dann liegt die ganze Folge in der Kugel \(B_R(x_{k^*})\), also ist \(\{x_k:k\in\mathbb{N}\}\) beschränkt.

    \item[(iii)] Sei \(x\) ein Häufungs- bzw. Anhäufungspunkt (H.P.) der Cauchy-Folge \((x_k)_{k\in\mathbb{N}}\) und sei \(\varepsilon>0\) beliebig. Da \((x_k)\) Cauchy ist, existiert \(k^*\in\mathbb{N}\) mit
    \[
    d(x_k,x_n)<\frac{\varepsilon}{2}\qquad\text{für alle }k,n\ge k^*.
    \]
    Weil \(x\) Häufungswert der Folge ist, gibt es ein \(m^*\ge k^*\) mit
    \[
    d(x,x_{m^*})<\frac{\varepsilon}{2}.
    \]
    Für alle \(k\ge k^*\) folgt dann
    \[
    d(x_k,x)\le d(x_k,x_{m^*})+d(x_{m^*},x)<\frac{\varepsilon}{2}+\frac{\varepsilon}{2}=\varepsilon.
    \]
    Damit gilt \(x_k\to x\). \(\square\)
    \end{enumerate}
\end{beweisbox}

\begin{definition}{Vollständiger metrischer Raum}
    Ein metrischer Raum heißt \emph{vollständig}, wenn jede Cauchy-Folge in ihm konvergiert.
\end{definition}

\begin{korollar}{Vollständigkeit und Cauchy-Folgen}
    In einem vollständigen metrischen Raum ist eine Folge genau dann eine Cauchy-Folge, wenn sie konvergiert.
\end{korollar}

\begin{beweisbox}
    Folgt aus Eigenschaften von C.F. (Satz 2.1.13 (i)) und Definition 2.1.14. q.e.d.
\end{beweisbox}

\begin{bemerkungbox}
    \[
    (X,d)\ \text{vollständig} \Longleftrightarrow (\text{C.F.}\Rightarrow \text{Konvergenz})
    \]
\end{bemerkungbox}

\begin{satz}{Vollständigkeit von $\mathbb{R}^n$}
    Der Raum $\mathbb{R}^n$ ist vollständig. Als normierter Raum ist er daher ein Banachraum.
\end{satz}

\begin{beweisbox}
    Sei $(x_k)_{k\in\mathbb{N}}\subset\mathbb{R}^n$ C.F.\\

    Zu zeigen: Es existiert $a\in\mathbb{R}^n$ mit $x_k\to a$ für $k\to\infty$.\\

    Betrachte die $i$-te Komponente $x_{k,i}$ von $x_k$. Für alle $i=1,\dots,n$ gilt
    \[
    |x_{k,i}-x_{m,i}|\leq\|x_k-x_m\|_\infty\xrightarrow[k,m\to\infty]{}0,
    \]
    da $(x_k)_{k\in\mathbb{N}}$ eine Cauchy-Folge in $\mathbb{R}^n$ ist. Daher ist für jedes feste $i$ die Folge $(x_{k,i})_{k\in\mathbb{N}}$ eine Cauchy-Folge in $\mathbb{R}$. Da $\mathbb{R}$ vollständig ist, existiert für jedes $i$ ein Grenzwert $a_i\in\mathbb{R}$ mit
    \[
    x_{k,i}\xrightarrow[k\to\infty]{}a_i.
    \]

    Damit konvergieren alle Komponenten von $x_k$ und es folgt die komponentenweise Konvergenz
    \[
    x_k\xrightarrow[k\to\infty]{}a:= (a_1,\dots,a_n)\in\mathbb{R}^n.
    \]
    \hfill $\square$   
\end{beweisbox}
